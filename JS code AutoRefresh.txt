 <script language="javascript">
		
		setInterval("AutoRefresh()", 1000)
		var reloadTime = 60;
		var counter = 0;
		function AutoRefresh()
		{
			window.status = "Refreshing page in " + ( reloadTime - counter) + " seconds ";
			for (i=0;i<counter%5;i++)
				window.status += '.';
			
			counter++;	
			if (counter > reloadTime)
			{
				counter = 0;
				window.status = "Refreshing page now, please wait...";
				window.location.reload();
				
			}
		}
		
    </script>
